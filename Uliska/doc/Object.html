def to_num

  number = /^([-+]?\d+(\.\d+)?)(\s?[GMK])?$/

  return self unless self.is_a? String
  return self unless self =~ number 

  decimal, suffix = $2, $3

  self.sub!(number,'\1')

  coeff = case suffix
          when /G/ then 1073741824 
          when /M/ then 1048576 
          when /K/ then 1024
          end

  (coeff || 1) * (decimal ? self.to_f : self.to_i)
  
end
