def initialize
  interval = [::DEFAULT[:ttl],1].max || nil
  if interval && ::DEFAULT[:start_background_parser]
    puts "[#{Time.now}] Starting background parser thread with interval #{interval} sec"
    $bg = Thread.new {
      loop {
        $nagios[:status].parse
        sleep interval
      } #loop
    } # thread
  end
end
