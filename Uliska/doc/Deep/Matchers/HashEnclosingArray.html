def matches? target

  @target = target
  result = true
  @failure = ":\n"
  
  if @target.is_a? Hash
    target.each do |key,array|
      if array.is_a? Array
        res = KeyNamesOfEnclosedHash.new(key).matches?(array)
        result &&= res
        @failure << { key => array}.inspect unless res
      end
    end
  end
  
  
  result
end
